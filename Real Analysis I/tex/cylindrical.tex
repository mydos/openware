\section{Cylindrical Coordinates} % (fold)
\label{sec:cylindrical}

\begin{defn}[Cylindrical Coordinates]
	The $\emph{cylindrical coordinates}$ are defined
	to be the polar map with z-coordinate. That is,
	\[
		\phi(r,\theta,z) =
		\begin{pmatrix}
			r \cos(\theta), & r \sin(\theta), & z
		\end{pmatrix}
	\]
\end{defn}
